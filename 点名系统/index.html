    const studentContainer = document.getElementById('student-container');
    const calledList = document.getElementById('called-list');
    const randomStudent = document.getElementById('random-student');
    const callRandomBtn = document.getElementById('call-random');
    const resetAllBtn = document.getElementById('reset-all');

    let currentSelectedStudent = null;

    // 渲染学生列表
    function renderStudents() {
        studentContainer.innerHTML = '';
        students.forEach(student => {
            const card = document.createElement('div');
            card.className = `student-card ${student.called ? 'called' : ''}`;
            card.dataset.id = student.id;
            card.innerHTML = `

            `;
            studentContainer.appendChild(card);
        });
        
        updateStats();
    }
    
    // 更新统计信息
    function updateStats() {
        const total = students.length;
        const calledCount = students.filter(s => s.called).length;
        const uncalledCount = total - calledCount;
        const participationRate = Math.round((calledCount / total) * 100);
        
        document.getElementById('total-students').textContent = total;
        document.getElementById('called-count').textContent = calledCount;
        document.getElementById('uncalled-count').textContent = uncalledCount;
        document.getElementById('participation-rate').textContent = `${participationRate}%`;
    }
    
    // 随机选择动画效果
    function startSelectionAnimation() {
        // 清除之前的动画
        if (highlightInterval) clearInterval(highlightInterval);
        
        // 获取所有未点名的学生卡片
        const uncalledCards = Array.from(studentContainer.children).filter(card => {
            const id = card.dataset.id;
            const student = students.find(s => s.id === id);
            return student && !student.called;
        });
        
        if (uncalledCards.length === 0) return;
        
        let currentIndex = 0;
        
        // 先高亮所有卡片一次
        uncalledCards.forEach((card, index) => {
            setTimeout(() => {
                card.classList.add('highlight');
                setTimeout(() => card.classList.remove('highlight'), 200);
            }, index * 50);
        });
        
        // 然后随机选择几个卡片高亮
        setTimeout(() => {
            highlightInterval = setInterval(() => {
                // 移除之前的高亮
                uncalledCards.forEach(card => card.classList.remove('highlight'));
                
                // 随机选择一个新的卡片高亮
                const randomIndex = Math.floor(Math.random() * uncalledCards.length);
                uncalledCards[randomIndex].classList.add('highlight');
                
                currentIndex = randomIndex;
            }, 200);
        }, uncalledCards.length * 50 + 300);
        
        return uncalledCards;
    }
    
    // 停止选择动画并选择最终学生
    function stopSelectionAnimation(uncalledCards, selectedStudent) {
        if (highlightInterval) clearInterval(highlightInterval);
        
        // 找到选中学生的卡片
        const selectedCard = uncalledCards.find(card => card.dataset.id === selectedStudent.id);
        
        if (selectedCard) {
            // 移除所有高亮
            uncalledCards.forEach(card => card.classList.remove('highlight'));
            
            // 高亮选中的卡片
            selectedCard.classList.add('highlight');
            
            // 标记为已点名
            selectedStudent.called = true;
            
            // 更新显示
            renderStudents();
            
            // 显示随机选择的学生
            randomStudent.innerHTML = `
                <div class="random-avatar pulse">
                    <i class="fas fa-user"></i>
                </div>
                <div class="random-name">${selectedStudent.name}</div>
                <div class="random-id">${selectedStudent.id}</div>
            `;
            
            // 添加到最近点名列表
            const calledItem = document.createElement('div');
            calledItem.className = 'called-item';
            calledItem.innerHTML = `
                <span>${selectedStudent.name} (${selectedStudent.id})</span>
                <span>${new Date().toLocaleTimeString()}</span>
            `;
            
            // 如果第一条是提示信息，移除它
            if (calledList.firstChild.textContent.includes('暂无点名记录')) {
                calledList.innerHTML = '';
            }
            
            calledList.prepend(calledItem);
            
            // 显示提示框
            modalName.textContent = selectedStudent.name;
            modalId.textContent = `学号：${selectedStudent.id}`;
            modal.style.display = 'flex';
            
            currentSelectedStudent = selectedStudent;
        }
    }
    
    // 随机点名功能
    callRandomBtn.addEventListener('click', () => {
        // 获取未点名的学生
        const uncalledStudents = students.filter(student => !student.called);
        
        if (uncalledStudents.length === 0) {
            alert('所有学生都已被点名！');
            return;
        }
        
        // 开始选择动画
        const uncalledCards = startSelectionAnimation();
        
        // 2秒后停止动画并选择学生
        setTimeout(() => {
            // 随机选择一名学生
            const randomIndex = Math.floor(Math.random() * uncalledStudents.length);
            const selectedStudent = uncalledStudents[randomIndex];
            
            // 停止动画并选择学生
            stopSelectionAnimation(uncalledCards, selectedStudent);
        }, 2000);
    });
    
    // 再点一位功能
    modalAnother.addEventListener('click', () => {
        modal.style.display = 'none';
